// refer method
"g d" to refer method

// rename
"c d" or "." edit variable name

// refactor & option
"cmd ." list option

// move
"num j" down
"num k" up
gg " Go to first line
G " Go to last line
Ctrl+g " Show current line info
"[ m" jump function up to down
"] m" jump function down to up

// block
"[]" find last of block
"v i a" selector all inside block
"v a t" selector block to last cursor
"v i t " find children and selector all inside
"v i i " selector children content inside block

// parameter
"v a a" selector parameter

// select same word
"g L" selector same word previous
"g l" selector same word
"g A" make a cursor
"g I" selector all block
"g a" select same word all **
"I" cursor to be front of word **
"A" cursor to be back of word

//
"ctr k" delete line
"ctr h" delete word from left to right
